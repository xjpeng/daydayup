#排序算法
- 基本算法
  - 冒泡排序
  - 选择排序
  - 插入排序

```javascript
 //冒泡排序
   function bubblesort(arr){
	   var len = arr.length;
	   var tmp,i,j,change;
	   for(i=0;i<len-1;i++){ //循环总次数
		  change=0;
		  for(j=0;j<len-i-1;j++){ //内循环,循环次数为总数-外循环次数
			 if(arr[j]<arr[j+1]){ //比较相邻两个元素
				tmp = arr[j];
				arr[j] = arr[j+1];
				arr[j+1] = tmp;
				change=1;
			 }
		  }
		  if(!change) break; //如果没有变化,就不用排序了
	   }
	   return arr;
   }
```

 //选择排序
function selectionsort(arr){
      var len = arr.length;
	  var max,tmp;
      for(var i=0;i<len-1;i++){
	     max = i;
		 for(var j=i+1;j<len;j++){
		    if(arr[max]<arr[j]){
			  max = j;
			}
		 }
		 if(max!=i){
		    tmp = arr[i];
			arr[i] = arr[max];
			arr[max] = tmp;
		 }
	  }
	  return arr;
